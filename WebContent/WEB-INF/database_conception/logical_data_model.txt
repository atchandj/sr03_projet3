	#id: INT,
	name: VARCHAR(50)

Adress(
	#id: INT,
	street: VARCHAR(100),
	town: VARCHAR(50),
	postCode: VARCHAR(5)
) avec KEY(street, town, postCode)

/* Ajout d'une clé artificielle pour les performances */

F+ = {
	id -> (street, town, postCode);
	(street, town, postCode) -> id;

	yearBook=>YearBook,
	name: VARCHAR(50),
	phone: VARCHAR(10),
	category=>Category,
